function TEmpresaLicencaManager.GetCNPJEmpresaAtual: string;
begin
  // Retornar CNPJ da empresa carregada em dados.qryEmpresa
  if Assigned(dados) and dados.qryEmpresa.Active and not dados.qryEmpresa.IsEmpty then
    Result := dados.qryEmpresaCNPJ.AsString
  else
    Result := '';
end;
